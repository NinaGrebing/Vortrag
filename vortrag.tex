\begin{frame}
	\frametitle{Aufgabenstellung}
	\begin{itemize}
		\item Simulation und Auswertung des Kundenverhaltens eines Carsharing Unternehmens
		\item Stadt in m $\times$ m Teilgebiete unterteilt
		\item Verlauf von Nachfrage und Abstellung der Autos für jedes Gebiet als monoton wachsende Polynome maximal 4. Grades, positiv auf dem Intervall [0;24]
		\item Einlesen der Daten aus einer Datei
		\item Ermitteln der Bedarfsänderungen für jedes Gebiet
		\begin{itemize}
			\item Integrale der Polynome bilden
			\item Ganzzahlübergänge durch Bisektion näherungsweise bestimmen
		\end{itemize}
	\end{itemize}
	 
	
\end{frame}

			\item maximaler Bedarf und Endzustand
		\end{itemize}
	\end{itemize}
	\begin{block}{Präzisierungen}
		\begin{itemize}
			\item Minimum 1 Gebiet
			\item Maximum 10 Gebiete
			\item Abbruchgenauigkeit der Bisektion: $e < 10^{-4}$
		\end{itemize}
	\end{block}
	
\end{frame}

\begin{frame}
	\frametitle{Berechnung der Bedarfsänderungen}
	\begin{itemize}
		\item Berechnung für jedes Teilgebiet
		\begin{itemize}
			\item Berechnung für Nachfragen und Abstellung
			\item Nullstellenproblem (nächster ganzzahliger Wert wird vom Funktionswert subtrahiert)
			\begin{itemize}			
				\item Bisektion im Intervall [$t_0$,$t_1$]
				\item $t_0$, beginnend mit 0, Stelle der letzten Änderung
				\item $t_1$ = 24
			\end{itemize}
			\item Änderungen der Nachfragen und Abstellungen zusammenfügen und zeitlich sortieren
		\end{itemize}
	\item Änderungen aller Teilgebiete zusammenfügen
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Bisektion}
	\begin{columns}
		\begin{column}{0.55\textwidth}
			\begin{itemize}
				\item bestimme y = Funktionswert an der Stelle x - nächster ganzzahliger Wert
				\item x = $t_0$ + $\frac{t_1-t_0}{2}$
				\item Abbruch bei:
				\begin{itemize}
					\item |y| < e = $10^{-4}$
					\item gebe x zurück
				\end{itemize}
				\item ansonsten Bisektion erneut durchführen
				\begin{itemize}
					\item y < 0: Bisektion für $t_0 = x, t_1 = t_1$
					\item y > 0: Bisektion für $t_0 = t_0, t_1 = x$
				\end{itemize}
			\end{itemize}
		\end{column}
		\begin{column}{0.45\textwidth}
			\begin{figure}
				\includegraphics[width=5cm]{./bisektion.jpg}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}
	\frametitle{Teststrategie}
	\begin{block}{White-Box-Tests}
		\begin{itemize}
			\item Test gegen die Implementierung
			\item alle Programmzweige werden durchlaufen
		\end{itemize}
	\end{block}
	\begin{block}{Testklassen}
		\begin{itemize}
			\item Normalfälle
			\item Sonderfälle
			\item Fehlerfälle
		\end{itemize}
	\end{block}	
\end{frame}

\begin{frame}
	\frametitle{Testklassen}
	\textbf{Normalfälle:}
	\begin{itemize}
		\item 2 <= m <= 10
		\item mindestens eine Nachfrage oder Abstellung in der gesamten Simulation
		\item höchstens 75 Nachfragen und 75 Abstellungen pro Teilgebiet
	\end{itemize}
	\textbf{Sonderfälle:}
	\begin{itemize}
		\item m = 1
		\item Nachfragen und Abstellungen für ein Teilgebiet gleich
		\item alle Teilgebiete sind äquivalent
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Testklassen}
	\textbf{Fehlerfälle:}
	\begin{itemize}
		\item m = 0
		\item m > 10
		\item Anzahl von Nachfragen/Abstellung für ein Teilgebiet > 75
		\item keine Bedarfsänderungen in der gesamten Simulation
		\item erste Zeile der Eingabedatei keine Kommentarzeile
		\item kein Zahlenwert für m
		\item kein Zahlenwert für Koeffizienten eines Polynoms
		\item zu wenig Polynome
		\item zu viele Nicht-Kommentarzeilen
		\item zu wenig Koeffizienten für ein Polynom
		\item zu viele Koeffizienten für ein Polynom
	\end{itemize}
\end{frame}
